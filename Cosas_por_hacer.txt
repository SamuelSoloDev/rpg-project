/* ------------------- HABILIDADES DEL DESTRUCTOR ------------------- */

const Golpe_Destructor = new Habilidad({
  nombre: "Golpe Destructor",
  efecto: (usuario, objetivo) => {
    objetivo.recibirDamageFisico(usuario.totalAtq * Golpe_Destructor.escalado, usuario);
  },
  tipo: "habilidad",
  naturaleza: "ofensivo",
  rango: "unico",
  precision: 95,
  escalado: 1.8 // daño muy alto a un solo objetivo
});

const Aniquilacion = new Habilidad({
  nombre: "Aniquilación",
  efecto: (usuario, objetivo) => {
    // daño masivo a todos
    objetivo.recibirDamageFisico(usuario.totalAtq * Aniquilacion.escalado, usuario);
  },
  tipo: "habilidad",
  naturaleza: "ofensivo",
  rango: "todos",
  precision: 90,
  escalado: 2.2 // daño altísimo
});

const Disipar_Rugido = new Habilidad({
  nombre: "Disipar Rugido",
  efecto: (usuario, objetivo) => {
    // elimina todos los buffs activos
    objetivo.limpiarModificadores("buff");
  },
  tipo: "habilidad",
  naturaleza: "debuff",
  rango: "todos",
  precision: 100
});

// Pasiva: se implementa con signals
const SED_DE_SANGRE = {
  nombre: "Sed de Sangre",
  trigger: "onCriticalReceived",
  efecto: (usuario) => {
    const buff = new StatModifier({
      nombre: "Sed de Sangre ATQ",
      duracion: 3,
      valorEfecto: 0.1, // +10% ATQ
      ritmo: "inicio",
      tipo: "buff",
      stat: "atq"
    });
    buff.aplicarModificador(usuario);
  }
};
